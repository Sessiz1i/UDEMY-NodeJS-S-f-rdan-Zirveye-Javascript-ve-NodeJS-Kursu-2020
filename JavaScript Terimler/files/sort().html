Dizi elemanlarını sıralama. Rakkam lar ve alfabe için sıralama yöntemleri aşağıdaki gibi dir

    // Numeric Sıralama
    const sayilar = [1, 50, 2, 41, 41, 84, 8, 35, 3]
    sayilar.sort((a,z) => a-z);
                    [1, 2, 3, 8, 35, 41, 41, 50, 84]

    // Alfabetik Sıralama ( Türkçe Karakterler ile )
    const isimler = ["Zeynel","Ümit","İlter","Ömer","Ali","Veli","Osman","Bekir"];
    isimler.sort((a, z) => a.localeCompare(z))
                    ["Ali","Bekir","İlter","Osman","Ömer","Ümit","Veli","Zeynel"];
