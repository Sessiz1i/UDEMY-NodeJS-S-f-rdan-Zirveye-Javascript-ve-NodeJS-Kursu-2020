    function bgRenkDegis(e) {
        document.querySelector(".yazi")
            .textContent = `R: ${e.clientX % 255} G: ${e.clientY % 255} B: ${e.clientY + e.clientX % 255}`;
        document.querySelector('.container')
            .style.backgroundColor = `rgb(${e.clientX % 255},${e.clientY % 255},${e.clientY + e.clientX % 255})`;

    }
